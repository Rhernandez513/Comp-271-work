bool ContainsDuplicates(String[] strings)
{
     for(int  n= 0; n < strings.Length; n++)
     {
          for(int m = 0; m < strings.Length; m++)
          {
               if(n == m) // Don't compare with self
               {
                    continue;
               }

               if(strings[n] == strings[m])
               {
                    return true;
               }
          }
     }
     return false;
}
